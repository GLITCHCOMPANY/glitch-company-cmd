function Show-GlitchRandomMessage {

    try{

        $randomContent=(Invoke-WebRequest `
        -Uri $GLITCH_RANDOM_URL `
        -UseBasicParsing).Content

        $lines=$randomContent -split "`n"

        $clean=@()

        foreach($line in $lines){

            $line=$line.Trim()

            if($line.Length -gt 0){

                $clean+=$line
            }
        }

        if($clean.Count -gt 0){

            $msg=Get-Random $clean

            Write-Host ""
            Write-Host "RANDOM PHRASE OF THE DAY!: "
            Write-Host $msg
            Write-Host ""
        }

    }catch{

        Write-Host "[RANDOM MESSAGE FAILED]"
    }
}

function Start-GlitchChat {

    try{

        if(!(Test-GlitchAuth)){ return }

        $chatDir="$env:USERPROFILE\glitch_chat"

        if(!(Test-Path $chatDir)){

            New-Item `
            -ItemType Directory `
            -Path $chatDir `
            -Force | Out-Null
        }

        $room=Read-Host "ROOM ID"

        if([string]::IsNullOrWhiteSpace($room)){

            Write-Host "[INVALID ROOM]"
            return
        }

        $chatFile="$chatDir\$room.txt"

        if(!(Test-Path $chatFile)){

            New-Item `
            -ItemType File `
            -Path $chatFile `
            -Force | Out-Null
        }

        while($true){

            Clear-Host

            Write-Host "=========================="
            Write-Host "GLITCH PRIVATE CHAT"
            Write-Host "ROOM: $room"
            Write-Host "ID: $global:GLITCH_CHAT_ID"
            Write-Host "=========================="
            Write-Host ""

            try{

                $content=Get-Content `
                $chatFile `
                -ErrorAction SilentlyContinue

                foreach($line in $content){

                    Write-Host $line
                }

            }catch{}

            Write-Host ""
            Write-Host "[ENTER = REFRESH]"
            Write-Host "[TYPE exit TO CLOSE]"
            Write-Host ""

            $msg=Read-Host "MESSAGE"

            if($msg -eq "exit"){

                break
            }

            if([string]::IsNullOrWhiteSpace($msg)){

                continue
            }

            $time=Get-Date -Format "HH:mm:ss"

            Add-Content `
            $chatFile `
            "[$time][$global:GLITCH_CHAT_ID] $msg"

            Start-Sleep -Milliseconds 300
        }

    }catch{

        Write-Host "[CHAT FAILED]"
    }
}

function Test-GitHubAuth {

    $token=Read-Host "GITHUB TOKEN"

    if([string]::IsNullOrWhiteSpace($token)){

        Write-Host "[EMPTY TOKEN]"

        return $false
    }

    $headers=@{
        Authorization="token $token"
        Accept="application/vnd.github+json"
    }

    try{

        $user=Invoke-RestMethod `
        -Uri "https://api.github.com/user" `
        -Headers $headers

        $global:GLITCH_GITHUB_USER=$user.login

        Write-Host ""
        Write-Host "[AUTH OK]"
        Write-Host "USER: $($user.login)"
        Write-Host ""

        return $true

    }catch{

        Write-Host "[AUTH FAILED]"

        return $false
    }
}
