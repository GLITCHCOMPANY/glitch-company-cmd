function Start-GlitchSearch {

    param([string]$Query)

    Play-GlitchBeep 800 60

    try{

        $url="https://html.duckduckgo.com/html/?q=$Query"

        $headers=@{
            "User-Agent"="Mozilla/5.0"
        }

        $html=(Invoke-WebRequest `
        -Uri $url `
        -Headers $headers `
        -UseBasicParsing).Content

    }catch{

        Write-Host "[SEARCH FAILED]"

        Write-GlitchError $_

        return
    }

    $matches=[regex]::Matches(
        $html,
        'result__a.*?href="(.*?)".*?>(.*?)</a>'
    )

    if($matches.Count -eq 0){

        Write-Host "[NO RESULTS]"
        return
    }

    $results=@()

    Write-Host ""
    Write-Host "========== SEARCH RESULTS =========="
    Write-Host ""

    foreach($m in $matches){

        $title=$m.Groups[2].Value

        $link=[System.Web.HttpUtility]::UrlDecode(
            $m.Groups[1].Value
        )

        $title=$title -replace '<.*?>',''

        if(
            $title.Length -gt 0 `
            -and `
            $results.Count -lt 10
        ){

            Write-Host "[" + $results.Count + "]"
            Write-Host $title
            Write-Host ""

            $results += $link
        }
    }

    if($results.Count -eq 0){

        Write-Host "[NO RESULTS]"
        return
    }

    Write-Host ""

    $choice=Read-Host "SELECT RESULT"

    if($choice -notmatch '^\d+$'){

        return
    }

    $index=[int]$choice

    if($index -ge $results.Count){

        return
    }

    $target=$results[$index]

    Write-Host ""
    Write-Host "========== WEBSITE CONTENT =========="
    Write-Host ""

    try{

        $site=(Invoke-WebRequest `
        -Uri $target `
        -Headers $headers `
        -UseBasicParsing).Content

        $site=$site -replace '<script[\s\S]*?</script>',''
        $site=$site -replace '<style[\s\S]*?</style>',''
        $site=$site -replace '<.*?>',' '
        $site=$site -replace '&nbsp;',' '
        $site=$site -replace '\s+',' '

        if($site.Length -gt 8000){

            $site=$site.Substring(0,8000)
        }

        Write-Host $site

    }catch{

        Write-Host "[FAILED TO LOAD WEBSITE]"

        Write-GlitchError $_
    }
}

function Start-GlitchChat {

    try{

        if(!(Test-GlitchAuth)){ return }

        $chatDir="$env:TEMP\glitch_chat"

        if(!(Test-Path $chatDir)){

            New-Item `
            -ItemType Directory `
            -Path $chatDir `
            -Force | Out-Null
        }

        $room=Read-Host "ROOM ID"

        if([string]::IsNullOrWhiteSpace($room)){

            Write-Host "[INVALID ROOM]"
            return
        }

        $chatFile="$chatDir\$room.txt"

        if(!(Test-Path $chatFile)){

            New-Item `
            -ItemType File `
            -Path $chatFile `
            -Force | Out-Null
        }

        while($true){

            Clear-Host

            Write-Host "=========================="
            Write-Host "GLITCH PRIVATE CHAT"
            Write-Host "ROOM: $room"
            Write-Host "ID: $global:GLITCH_CHAT_ID"
            Write-Host "=========================="
            Write-Host ""

            Get-Content `
            $chatFile `
            -ErrorAction SilentlyContinue

            Write-Host ""
            Write-Host "[TYPE exit TO CLOSE]"
            Write-Host ""

            $msg=Read-Host "MESSAGE"

            if($msg -eq "exit"){

                break
            }

            if($msg.Length -eq 0){

                continue
            }

            $time=Get-Date -Format "HH:mm:ss"

            Add-Content `
            $chatFile `
            "[$time][$global:GLITCH_CHAT_ID] $msg"
        }

    }catch{

        Write-Host "[CHAT FAILED]"

        Write-GlitchError $_
    }
}

function Start-GlitchUpdate {

            Write-Host "[UPDATING TO LATEST VERSION]"

            $url="https://raw.githubusercontent.com/GLITCHCOMPANY/glitch-company-cmd/main/code.txt?t=$(Get-Random)"

        }else{

            Write-Host "[UPDATING TO VERSION $Version]"

            $url="https://raw.githubusercontent.com/GLITCHCOMPANY/glitch-company-cmd/refs/tags/$Version/code.txt?t=$(Get-Random)"
        }

        $temp="$env:TEMP\glitch_install.ps1"

        Remove-Item `
        $temp `
        -Force `
        -ErrorAction SilentlyContinue

        Invoke-WebRequest `
        -Uri $url `
        -UseBasicParsing `
        -OutFile $temp

        if(!(Test-Path $temp)){

            Write-Host "[UPDATE FILE FAILED]"
            return

        powershell `
        -ExecutionPolicy Bypass `
        -NoExit `
        -File $temp

        Write-Host "[UPDATED SUCCESSFULLY]"

    }catch{

        Write-Host "[UPDATE FAILED]"

        Write-GlitchError $_
    }
}
