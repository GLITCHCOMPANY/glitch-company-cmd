function Write-GlitchError {

    param($ErrorMessage)

    try{

        $time=Get-Date -Format "yyyy-MM-dd HH:mm:ss"

        $log="[$time] $ErrorMessage"

        Add-Content `
        "$env:TEMP\glitch_error_logs.txt" `
        $log

    }catch{}
}

function Publish-GlitchLogs {

    try{

        if([string]::IsNullOrWhiteSpace($GLITCH_GITHUB_TOKEN)){

            Write-Host "[NO GITHUB TOKEN]"
            return
        }

        $logFile="$env:TEMP\glitch_error_logs.txt"

        if(!(Test-Path $logFile)){

            Write-Host "[NO LOG FILE]"
            return
        }

        $content=Get-Content $logFile -Raw

        $base64=[Convert]::ToBase64String(
            [Text.Encoding]::UTF8.GetBytes($content)
        )

        $headers=@{
            Authorization="token $GLITCH_GITHUB_TOKEN"
            Accept="application/vnd.github+json"
        }

        $api="https://api.github.com/repos/$GLITCH_REPO/contents/ERROR_LOGS"

        try{

            $old=Invoke-RestMethod `
            -Uri $api `
            -Headers $headers `
            -Method Get

            $sha=$old.sha

        }catch{

            $sha=$null
        }

        $body=@{
            message="update error logs"
            content=$base64
            sha=$sha
        } | ConvertTo-Json

        Invoke-RestMethod `
        -Uri $api `
        -Headers $headers `
        -Method Put `
        -Body $body

        Write-Host "[ERROR LOGS PUBLISHED]"

    }catch{

        Write-Host "[FAILED TO PUBLISH LOGS]"

        Write-GlitchError $_
    }
}

            Write-Host $msg

        }

    }catch{

        Write-Host "[RANDOM MESSAGE FAILED]"
    }
}

function Start-GlitchChat {

    try{

        if(!(Test-GlitchAuth)){ return }

        $chatDir="$env:TEMP\glitch_chat"

        if(!(Test-Path $chatDir)){

            New-Item `
            -ItemType Directory `
            -Path $chatDir `
            -Force | Out-Null
        }

        $room=Read-Host "ROOM ID"

        if([string]::IsNullOrWhiteSpace($room)){

            Write-Host "[INVALID ROOM]"
            return
        }

        $chatFile="$chatDir\$room.txt"

        if(!(Test-Path $chatFile)){

            New-Item `
            -ItemType File `
            -Path $chatFile `
            -Force | Out-Null
        }

        while($true){

            Clear-Host

            Write-Host "=========================="
            Write-Host "GLITCH PRIVATE CHAT"
            Write-Host "ROOM: $room"
            Write-Host "ID: $global:GLITCH_CHAT_ID"
            Write-Host "=========================="
            Write-Host ""

            try{

                $content=Get-Content `
                $chatFile `
                -ErrorAction SilentlyContinue

                foreach($line in $content){

                    Write-Host $line
                }

            }catch{}

            Write-Host ""
            Write-Host "[ENTER = REFRESH]"
            Write-Host "[TYPE exit TO CLOSE]"
            Write-Host ""

            $msg=Read-Host "MESSAGE"

            if($msg -eq "exit"){

                break
            }

            if([string]::IsNullOrWhiteSpace($msg)){

                continue
            }

            $time=Get-Date -Format "HH:mm:ss"

            Add-Content `
            $chatFile `
            "[$time][$global:GLITCH_CHAT_ID] $msg"

            Start-Sleep -Milliseconds 300
        }

    }catch{

        Write-Host "[CHAT FAILED]"

        Write-GlitchError $_
    }
}

function Start-GlitchUpdate {

    param([string]$Version)

    try{

        if([string]::IsNullOrWhiteSpace($Version)){

            Write-Host "[UPDATING TO LATEST VERSION]"

            $url="https://raw.githubusercontent.com/GLITCHCOMPANY/glitch-company-cmd/main/code.txt?t=$(Get-Random)"

        }else{

            Write-Host "[UPDATING TO VERSION $Version]"

            $url="https://raw.githubusercontent.com/GLITCHCOMPANY/glitch-company-cmd/releases/download/$Version/code.txt?t=$(Get-Random)"
        }

        $temp="$env:TEMP\glitch_install.ps1"

        Remove-Item `
        $temp `
        -Force `
        -ErrorAction SilentlyContinue

        Invoke-WebRequest `
        -Uri $url `
        -UseBasicParsing `
        -OutFile $temp

        if(!(Test-Path $temp)){

            Write-Host "[VERSION NOT FOUND]"
            return
        }

        $content=Get-Content $temp -Raw

        if($content.Length -lt 20){

            Write-Host "[INVALID UPDATE FILE]"
            return
        }

        Write-Host "[INSTALLING UPDATE]"

        powershell `
        -ExecutionPolicy Bypass `
        -NoExit `
        -File $temp

    }catch{

        Write-Host "[UPDATE FAILED]"

        Write-Host $_

        Write-GlitchError $_
    }
}
