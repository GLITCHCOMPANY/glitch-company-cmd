$GLITCH_VERSION="1.1.7"

# AUTO NEXT VERSION
$splitVersion=$GLITCH_VERSION.Split(".")

$major=[int]$splitVersion[0]
$minor=[int]$splitVersion[1]
$patch=[int]$splitVersion[2]

$patch++

$versionnext="$major.$minor.$patch"

$global:GLITCH_SOUND=$false

$identity=[Security.Principal.WindowsIdentity]::GetCurrent()

$principal=New-Object Security.Principal.WindowsPrincipal($identity)

$isAdmin=$principal.IsInRole(
    [Security.Principal.WindowsBuiltInRole]::Administrator
)

if(-not $isAdmin){

    Write-Host "[ADMIN REQUIRED]"
    Write-Host "REQUESTING ELEVATION..."

    Start-Process powershell `
    -Verb RunAs `
    -ArgumentList "-ExecutionPolicy Bypass -NoExit -Command `"Invoke-WebRequest 'https://raw.githubusercontent.com/GLITCHCOMPANY/glitch-company-cmd/main/code.txt?t=$(Get-Random)' -OutFile `$env:TEMP\glitch_admin.ps1; . `$env:TEMP\glitch_admin.ps1`""

    exit
}

function Play-GlitchBeep {

    param(
        [int]$freq,
        [int]$time
    )

    if($global:GLITCH_SOUND){

        [console]::beep($freq,$time)
    }
}

Play-GlitchBeep 900 120
Play-GlitchBeep 1200 120

Write-Host "=========================="
Write-Host "GLITCH COMPANY CMD"
Write-Host "VERSION $GLITCH_VERSION"
Write-Host "NEXT VERSION $versionnext"
Write-Host "=========================="

($env:TEMP + '\glitch_update.ps1'); Write-Host '[UPDATED SUCCESSFULLY]' }"

Write-Host "SYSTEM LOADED..."
